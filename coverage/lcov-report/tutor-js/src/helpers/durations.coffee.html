var TimeStore, _, moment, twix;
 
moment = require('moment');
 
twix = require('twix');
 
_ = require('underscore');
 
TimeStore = require('../flux/time').TimeStore;
 
module.exports = {
  create: function(startTime, endTime) {
    return moment(startTime).twix(endTime);
  },
  isPastDue: function(arg) {
    var due_at;
    due_at = arg.due_at;
    return moment(TimeStore.getNow()).isAfter(due_at);
  }
};
